        //set select

  Vue.component('formBottomPanel', {
    data() {
      return {};
    },
    computed: {
      checkedNum() {
        return this.$store.state.employees.filter(
          (employee) => employee.checked === true
        ).length;
      },
      num() {
        let word = this.declOfNum(this.checkedNum, [
          'выписка',
          'выписки',
          'выписок',
        ]);
        return `${this.checkedNum} ${word}`;
      },

    template: `
            <transition name="slide-fade">
                <div class="b-poll__form-bottom-panel" v-if="checkedNum > 0">
                    <div class="b-poll__form-bottom-panel__left">

                        </div>

                        <button class="btn btn-secondary" @click.prevent="clickBtn()">Сгенерировать выписки</button>
                    </div>
                </div>
            </transition>
        `,
    methods: {
      declOfNum(number, words) {
        return words[
          number % 100 > 4 && number % 100 < 20
            ? 2
            : [2, 0, 1, 1, 1, 2][number % 10 < 5 ? Math.abs(number) % 10 : 5]
        ];
      },
      clickBtn() {
        //show modal
        $('#getExcerptModal').modal('show');
        //show loading
        this.$store.commit('changeStatus', 'loading');
        //send request
        this.generate();
        //scroll to top
        window.scrollTo(0, 0);
        //reset checked
        this.$store.state.employees.forEach((employee, index) => {
          this.$store.commit('changeChecked', {
            checked: false,
            itemIndex: index,
          });
        });
      },
      generate() {
        let formData = new FormData(
          document.getElementById('excerptsMultipleGenerateForm')
        );
        (async () => {
          let counter = 0;
          do {
            try {
              let response;
              if (counter === 0) {
                response = await fetch(
                  window.excerptsMultipleGeneratePaths.response,
                  {
                    method: 'POST',
                    body: formData,
                  }
                );
              } else {
                response = await fetch(
                  window.excerptsMultipleGeneratePaths.onMounted
                );
              }
              counter++;
              let result = await response.json();

              if (result.STATUS === 'LOADING') {
                await new Promise((r) => setTimeout(r, 15000));
              } else if (result.STATUS !== 'Y') {
                //hide modal
                $('#getExcerptModal').modal('hide');
                //hide loading
                this.$store.commit('changeStatus', '');
                //throw error
                throw new Error('Ошибка');
              } else if (result.GENERATED) {
                setTimeout(() => {
                  //mutate store
                  this.$store.commit('setGenerated', result.GENERATED);
                  //hide modal
                  $('#getExcerptModal').modal('hide');
                  //hide loading
                  this.$store.commit('changeStatus', '');
                }, 500);
                return;
              } else {
                //hide modal
                $('#getExcerptModal').modal('hide');
                //hide loading
                this.$store.commit('changeStatus', '');
                break;
              }
            } catch (err) {
              throw err;
            }
          } while (true);
        })();
      },
    },
  });

        :class="propsClass"

        @click.stop.prevent="clickButton"
      >
        <icon-button-content :props="props" />
      </button>

      <button v-else-if="isButtonTag"
        class="vue-button"
        :class="propsClass"
        @click.prevent="clickButton"
      >
        <text-button-content :text="text" :props="props" />
      </button>

      <a v-else-if="isATag && isIcon"
        :class="propsClass"
        :title="text"
        :href="href"
      >
        <icon-button-content :props="props" />
      </a>

      <a v-else-if="isATag"
        class="vue-button"
        :class="propsClass"
        :href="href"
      >
        <text-button-content :text="text" :props="props" />
      </a>
    `,
    computed: {
      propsClass() {
        const result = {};
        if (this.props) {
          this.props.forEach((p) => {
            if (!p.startsWith('icon-')) {
              result[`vue-button--${p}`] = true;
            }
          });
        }

        if (this.disabled) {
          result[`vue-button--disabled`] = true;
        }
        return result;
      },
      isButtonTag() {
        return !this.href;
      },
      isATag() {
        return !!this.href;
      },
      isIcon() {
        return this.props.find(e => e === 'icon');
      },
    },
    methods: {
      clickButton() {
        this.$emit('clickButton');
      },
    }
  });

  Vue.component('docComponent', {
    data() {
      return {};
    },

        type: Object,

          id: 123,
          name: 'Протокол заседания дисицплинарной комиссии 234',

            <a v-for="(icon, index) in doc.icon" :key="index" class="twpx-doc__icon" :href="doc.href" :style="getStyle(index)" target="_blank"></a>
          </div>
          <span class="twpx-doc__text">
            <a v-if="doc.name" class="twpx-doc__name" :href="doc.href" target="_blank">{{ doc.name }}</a>
            <span v-if="doc.desc" class="twpx-doc__desc">{{ doc.desc }}</span>
            <span class="twpx-doc__data">
              <span v-if="doc.size && doc.href">{{ doc.size }} {{ getFileNameAndExt(doc.href).ext }}</span>
              <span v-if="doc.date">Создан: {{ formatDateToRussian(doc.date) }}</span>
              <span v-if="doc.author">Автор: {{ doc.author }}</span>
            </span>
          </span>
          <button-component v-if="doc.remove" :text="Delete" :props="['icon', 'delete', 'small']" @clickButton="$emit('clickDelete')" />
        </div>
      </div>
    `,
    computed: {},
    methods: {
      formatDateToRussian(dateString) {
        const months = [
          'января', 'февраля', 'марта', 'апреля', 'мая', 'июня',
          'июля', 'августа', 'сентября', 'октября', 'ноября', 'декабря'
        ];
        const date = new Date(dateString);
        if (isNaN(date)) return dateString; // если невалидная дата, возвращаем как есть
        const day = date.getDate();
        const month = months[date.getMonth()];
        const year = date.getFullYear();
        return `${day} ${month} ${year}`;
      },
      getStyle(index) {
        return `background-image: url(${this.doc.icon[index]})`;
      },
      getFileNameAndExt(path) {
        // Получаем только имя файла с расширением
        const fileNameWithExt = path.split('/').pop();
        // Находим позицию последней точки
        const lastDotIndex = fileNameWithExt.lastIndexOf('.');
        if (lastDotIndex === -1) {
          // Если точки нет, возвращаем всё как имя, расширение — пустая строка
          return { name: fileNameWithExt, ext: '' };
        }
        // Имя файла без расширения
        const name = fileNameWithExt.substring(0, lastDotIndex);
        // Расширение с точкой
        const ext = fileNameWithExt.substring(lastDotIndex);
        return { name, ext };
      },
      formatFileSize(bytes) {
        if (bytes === 0) return '0 б';
        const units = ['Б', 'Кб', 'Мб', 'Гб', 'Тб'];
        const k = 1024;
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        const size = bytes / Math.pow(k, i);
        // Для Мб и выше округляем до одного знака после запятой, для Кб и ниже — до целых
        if (i >= 2) {
          return `${size.toFixed(1)} ${units[i]}`;
        } else {
          return `${Math.round(size)} ${units[i]}`;
        }
      }
    },
  });

                        <form-bottom-panel></form-bottom-panel>
                    </form>

            if (result.STATUS === 'LOADING') {

            } else {

            throw err;

  const app = new Vue(App);

window.addEventListener('scroll', () => {
  //bottom panel select
  const panel = document.querySelector('.b-poll__form-bottom-panel');
  if (!panel) return;

  if (panel.getBoundingClientRect().top < 168 + 86) {
    panel.classList.add('i-bottom');
  } else {
    panel.classList.remove('i-bottom');
  }
});
